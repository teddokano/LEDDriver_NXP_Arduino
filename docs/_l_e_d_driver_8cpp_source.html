165void PCA995x_SPI::reg_access( uint8_t reg, uint8_t *vp, int len )
166{
167 uint8_t data[ len * 2 ];
168
169 for ( int i = 0; i < len; i++ ) {
170 data[ i * 2 + 0 ] = (reg << 1) + (i * 2);
